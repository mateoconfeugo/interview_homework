      host.vm.provision "ansible" do |ansible|
        ansible.extra_vars = { ENV: this_env }
        ansible.playbook = "ansible/homework.yml"
        ansible.groups = {
          "#{this_role}-#{this_env}" => [ "#{distribution}-#{this_role}" ]

    end
  end
